\begin{frame}{Data sets on emissions}

    \begin{wideitemize}
        \item[1.] \textbf{EUTL}: emissions and NACE activity by year for installations targeted by EUETS

        \item[2.] \textbf{National GHG Inventories}: emissions by CRT category by year

        \item[3.] \textbf{PEFA}: energy use of fuels by year and NACE category

        \item[4.] \textbf{AEA}: GHG emissions by year and NACE activity
    \end{wideitemize}
    
\end{frame}

\begin{frame}{}

\textbf{Goal}: emissions not covered by EUETS by sector-year

\begin{wideitemize}
    \item In principle, it should be simple: difference between AEA and EUTL
    \item However, those two quantities are not comparable
    \begin{wideitemize}
        \item[a.] AEA includes emissions from non-stationary sources;
        \item[b.] emissions in AEA for sector $s$ are proportional to sector $s$'s value added share among sectors mapped to a particular emission category in national inventory
    \end{wideitemize}
\end{wideitemize}
\end{frame}

\begin{frame}
\textbf{Alternative}: bottom-up approach

\begin{wideitemize}
    \item Fuel consumption (PEFA) + emissions from industrial process (national inventory)
    \begin{wideitemize}
        \item[] For sector $s$, fuel $f$: $z_{s,f} = q_{s,f} \times \text{EF}_{s,f}$
    \end{wideitemize}
    \item Problem: $q_{s,f}$ apparently includes quantity of fuel which is partially attributable to industrial process $\implies$ double-counting
\end{wideitemize}

\end{frame}

\begin{frame}
\textbf{Alternative}: PEFA and value added as weights on emissions from national inventory

\begin{wideitemize}
    \item Emissions for sector $s$:
    $$ Z_s = \sum_{c(s)} \sum_f \frac{q_{s,f}}{\sum_{r \in c(s)} q_{r,f}} \times z_{c(s),f} + \sum_{i(s)} \frac{VA_{s}}{\sum_{r \in i(s)} VA_{r}} \times z_{i(s)}$$
\end{wideitemize}

\end{frame}

\begin{frame}
\begin{wideitemize}
    \item \textbf{Example}: sector C20: Manufacture of chemicals
    \item 1A2c in inventory records GHG from fuel combustion from sectors C20, C21
    \item 2B records GHG from industrial processes in C20

    $$ Z_{\text{C20}} = \sum_f \frac{q_{\text{C20},f}}{q_{\text{C20},f} + q_{\text{C21},f}} \times z_{\text{1A2c,f}} + z_{\text{2B}}$$
\end{wideitemize}
\end{frame}

\begin{frame}
\begin{wideitemize}
    \item \textbf{Sanity check}: Inventory compares agg. and EUETS emissions for selected sectors for years 2022, 2023
\end{wideitemize}
\end{frame}
